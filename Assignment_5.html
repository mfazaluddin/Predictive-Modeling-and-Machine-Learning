# define- fit - predict - find the accuracy of models
from sklearn.metrics import accuracy_score
# linear kernel
svc=svm.SVC(kernel='linear', C=1).fit(x,y)
svc_pred=svc.predict(x)
print('svc accuracy',accuracy_score(y,svc_pred)*100)
# linear svm
lin_svc=svm.LinearSVC(C=1).fit(x,y)
lsvc_pred=lin_svc.predict(x)
print('linear svc accuracy',accuracy_score(y,lsvc_pred)*100)
# rbf kernel
rbf_svc=svm.SVC(kernel='rbf', gamma=0.7, C=1).fit(x,y)
rsvc_pred=rbf_svc.predict(x)
print('rbf accuracy',accuracy_score(y,rsvc_pred)*100)
# poly kernel, degree 3 ax^3+bx^2+cx+d
poly_svc=svm.SVC(kernel='poly', degree=3, C=1).fit(x,y)
psvc_pred=poly_svc.predict(x)
print('poly accuracy',accuracy_score(y,psvc_pred)*100)

h = 0.05
x_min, x_max = x[:, 0].min() - 1, x[:, 0].max() + 1  # Using NumPy array indexing
y_min, y_max = x[:, 1].min() - 1, x[:, 1].max() + 1
xx, yy = np.meshgrid(np.arange(x_min, x_max, h), np.arange(y_min, y_max, h))

# Create the plot
plt.figure(figsize=(15, 10))

titles = ['linear kernel', 'linear svc', 'rbf', 'poly']

for i, clf in enumerate((svc, lin_svc, rbf_svc, poly_svc)):
    # Create 4 subplots
    plt.subplot(2, 2, i + 1)
    plt.subplots_adjust(wspace=0.4, hspace=0.4)
    
    # Predict using the classifier
    Z = clf.predict(np.c_[xx.ravel(), yy.ravel()])
    Z = Z.reshape(xx.shape)
    
    # plot actual and predicted values
    plt.contourf(xx, yy, Z, cmap=plt.cm.coolwarm, alpha=0.8)
    plt.scatter(x[:, 0], x[:, 1], c=y, cmap=plt.cm.coolwarm)  # Using NumPy array indexing
    
    # Customize the plot
    plt.xlabel('Age and medication')
    plt.ylabel('Test result')
    plt.xlim(xx.min(), xx.max())
    plt.ylim(yy.min(), yy.max())
    plt.xticks(())
    plt.yticks(())
    plt.title(titles[i])

plt.show()

f) Interpret the graphs: Looking at these four models, the RBF shows the most realistic pattern by capturing complex relationships between age, medications, and test results - just like how in real medical cases, the interaction between patient age and drug response isn't always straightforward. The other three models seem too simplified for medical data, as they assume simple straight-line patterns. For healthcare predictions, we'd probably want to use the RBF model since it better reflects the complexity of real patient responses, even though it might look complex than the others. The striped pattern in the RBF plot suggests it's picking up on important variations in how different age groups respond to different medications.
